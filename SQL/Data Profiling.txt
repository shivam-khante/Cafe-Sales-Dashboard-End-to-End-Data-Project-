use cafe;

select * from dirty_cafe_sales;

create table bronze_cafe_sales as select * from dirty_cafe_sales;  -- Sample table to perform script

select 
sum(case when transaction_id is null or transaction_id='' then 1 else 0 end) as transaction_id, 
sum(case when item is null or item='' then 1 else 0 end) as item,
sum(case when quantity is null or quantity='' then 1 else 0 end) as quantity, 
sum(case when price_per_unit is null or price_per_unit='' then 1 else 0 end) as price_per_unit, 
sum(case when total_spent is null or total_spent='' then 1 else 0 end) as total_spent, 
sum(case when payment_method is null or payment_method='' then 1 else 0 end) as payment_method, 
sum(case when location is null or location='' then 1 else 0 end) as location,
sum(case when transaction_date is null or transaction_date='' then 1 else 0 end) as transaction_date
from bronze_cafe_sales;

select *
from (
    select 
        transaction_date,
        case 
            when transaction_date regexp '^[0-9]+-[0-9]{2}-[0-9]{2}' then 'correct'
            else 'incorrect' 
        end as date_format,
        count(*) as cnt
    from bronze_cafe_sales
    group by transaction_date
) t
where date_format = 'incorrect';
